import re
import os
def atoi(text):
    return int(text) if text.isdigit() else text

def natural_keys(text):
    return [ atoi(c) for c in re.split('(\d+)',text) ]
my_list =os.listdir()
my_list.sort(key=natural_keys)
print(my_list)
